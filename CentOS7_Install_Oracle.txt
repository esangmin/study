1. 디렉토리 생성 및 권한 변경
 - mkdir vm_ora12c
 - chmod -R a+w wm_ora12c
2. Docker Image 조회
 - docker search oracle
3. Docker Image Download
 - docker pull XXXX/xxxxxx
 - docker images
4. 실행
 - docker run --name oracle12c -d -p (외부포트):8080 -p (외부포트):1521 -v (외부경로):/u01/app/oracle XXXX/xxxxx
 - 예) docker run --name oracle12c -d -p 28080:8080 -p 21521:1521 -v
5. 실행 로그 확인
 - docker logs -f oracle12c
6. oracle 사용자 계정 생성
 - docker exec -it oracle12c /bin/bash
 - su oracle
 - export PATH=$PATH:$ORACLE_HOME/bin
 - sqlplus /as sysdba
 - create user testuser identified by oracle;
 - grant dba to testuser;
 - exit
 - exit
7. 해당 유저로 접속 확인
 - docker exec -it oracle12c sqlplus testuser/oracle
