○ 파일 설치 및 실행
 - Sonatype 홈페이지(https://www.sonatype.com/download-oss-sonatype)에서 OSS 3.X 버전을 다운로드하고 압축을 해제합니다
 - tar zxvf nexus-3.17.0-01-unix.tar.gz
 - nexus-3.17.0-01/bin/nexus run
 - 종료 : Ctrl + C

○ port 추가
 - sudo firewall-cmd --permanent --zone=public --add-port=8081/tcp
 - sudo firewall-cmd --reload

○ nexus 사용자 생성
 - sudo adduser nexus
 - sudo passwd nexus --> smcs
 - 

○ bin/nexus.rc 수정
 run_as_user="nexus"

○ sudo ln -s /home/centos7/nexus-3.17.0-01/bin/nexus /etc/init.d/nexus

○ chkconfig
 - $ cd /etc/init.d
 - $ sudo chkconfig --add nexus
 - $ sudo chkconfig --levels 345 nexus on
 - $ service nexus start
